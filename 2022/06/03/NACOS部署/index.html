一.前言

1.1简介

Nacos是阿里巴巴开源的一款支持服务注册与发现，配置管理以及微服务管理的组件。用来取代以前常用的注册中心（zookeeper , eureka等等），以及配置中心（spring cloud config等等）。Nacos是集成了注册中心和配置中心的功能，做到了二合一。

CREATE DATABASE IF NOT EXISTS nacos DEFAULT CHARSET utf8 COLLATE utf8_general_ci;

2.2 创建存储

kind: Deployment
apiVersion: apps/v1
metadata:
  name: nfs-provisioner-nacos
  namespace: kube-system
spec:
  replicas: 1
  strategy:
    type: Recreate
  selector:
    matchLabels:
      app: nfs-provisioner-nacos
  template:
    metadata:
      labels:
        app: nfs-provisioner-nacos
    spec:
      serviceAccountName: nfs-client-provisioner
      containers:
        - name: nfs-client-provisioner
          image: jmgao1983/nfs-client-provisioner:latest
          imagePullPolicy: IfNotPresent
          volumeMounts:
            - name: nfs-client-root
              mountPath: /persistentvolumes
          env:
            - name: PROVISIONER_NAME
              value: nfs-provisioner-nacos  # 此处供应者名字供storageclass调用
            - name: NFS_SERVER
              value: 10.46.8.18   # 填入NFS的地址
            - name: NFS_PATH
              value: /data/nacos_data  # 填入NFS服务器挂载的目录
      volumes:
        - name: nfs-client-root
          nfs:
            server: 10.46.8.18   # 填入NFS的地址
            path: /data/nacos_data   # 填入NFS服务器挂载的目录

---
apiVersion: storage.k8s.io/v1
kind: StorageClass
metadata:
  name: nfs-nacos
provisioner: nfs-provisioner-nacos
# Supported policies: Delete、 Retain ， default is Delete
reclaimPolicy: Retain
